select * from zh_en, bgfr_f where bgfr_f.bigram = zh_en.hans;

update zh_en set frqf = (select id from bgfr_f where bgfr_f.bigram = zh_en.hans);
select count(*) from zh_en where frqf is not null;

update zh_en set frqn = (select id from bgfr_n where bgfr_n.bigram = zh_en.hans);
select count(*) from zh_en where frqn is not null;

update zh_en set hans = hant where hans is null;
update zh_en set hans = NULL where hans = hant;
